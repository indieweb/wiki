// ==UserScript==
// @name         Instagram Comment Hook
// @match        https://www.instagram.com/*
// @grant        none
// ==/UserScript==

(function () {
  'use strict'

  document.addEventListener('click', function (e) {
    const postBtn = e.target.closest('div[role="button"]')

    if (postBtn && postBtn.innerText.trim() === 'Post') {
      const textarea = document.querySelector('textarea[aria-label="Add a comment…"]')
      if (textarea) {
        alert(textarea.value) // post to your website
      }
    }
  }, true)
})()
